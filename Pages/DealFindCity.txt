	var block_tag = null; //used in blockClick()
	
	function city_name()
	{
		var default_city_name = 'Select';
		var js_city_name;
		if($.session("sess_city_name"))
		{
			js_city_name = $.session("sess_city_name");
		}
		else
		{	
			js_city_name = default_city_name;
		}
		return js_city_name;
	}
	function set_city(id)
	{
		var default_city_name = '';
		// on select 
		if($.session("sess_city_name"))
		{
			if($.session("sess_city_name") == "Select")
			{
				$('#'+id).hide();
				$('#intro_logo').removeAttr("href");
			}
			else
			{
				$('#'+id).attr('href', $.session("sess_city_name"));
				$("#intro_logo").attr('href', $.session("sess_city_name"));
				$('#'+id).show();
			}
		}
		else
		{	
			// on first page load
			if(default_city_name == "")
			{
				$('#'+id).hide();
				$('#intro_logo').removeAttr("href");
			}
			else
			{
				$('#'+id).attr('href', default_city_name);
				$("#intro_logo").attr('href',default_city_name);
				$('#'+id).show();
			}
		}
	}
	$(document).ready(function() 
	{
		$('.step1').show();
		$('.step2').hide();
		$('.error1').hide();
		var val = $("#selectcity").val();
		if(val == "Select")
			$('#cityname').hide();

		{
			var val = $("#selectcity").val();
			// SET CITY AUTOMATIC TEXT TO HEADER 
			var city_name =  ($("#selectcity [value= "+val+"]").text());
			$.session("sess_city_name",val);
			set_city('skip_this');

			if(val == "Select")
			{
				$('#cityname').hide();
			}
			else
			{
				$("#cityname").html('<span class="intext">in</span>'+city_name);

			}

		var city_name1 = city_name();
		if(city_name1 =="Select")
		{
			$('#cityname').hide();
		}
		set_city('skip_this');
	});
	function step(hide, show)
	{
		var city_name1 = city_name();
		if(city_name1 == "Select")
		{
			$('.error1').show();
			return false;
		}
		else
		{
			$('.step'+hide).hide();
			$('.step'+show).show();
			$('.error').hide();
			$('.error1').hide();
			if(show == 2 )
			{
				$('#email').focus();
			}
		}
	}
	function IsValidEmail(email)
	{
		var filter = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w\-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
		return filter.test(email);
	}
	function validate()
	{
		if(!$('#email').val())
		{ 
			$('.error').show();
			$('#email').focus();
		}
		else if($('#email').val())
		{
			if (!IsValidEmail($('#email').val()))
			{
				$('.error').show();
				$('#email').focus();
				$('#continue2').attr('href', "#");
				return false;
			}
			else
			{
				//
				if(blockClick()){				
					$('.error').hide();
					document.frm_subscribe.submit();
				}
			}
		}
	}
	
	/* limits the amount of click a user can click a submit button 
	*  return true if the user can proceed
	*  return false if the user is blocked
	* 
	*  this function also exists in main.js
	*/
	function blockClick(){	
		var wait = 10000; //block clicking for a set time - in milliseonds
		var d = new Date();	
		
		if(block_tag == null || d.getTime() > block_tag){
			block_tag = d.getTime()+wait;
			//console.log('true');
			return true;
		}
		//console.log('false');
		return false;
	}
